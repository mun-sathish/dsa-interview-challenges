```js copy
/**
 * @param {number} rows
 * @param {number} cols
 * @param {number} rStart
 * @param {number} cStart
 * @return {number[][]}
 */
var spiralMatrixIII = function(rows, cols, rStart, cStart) {
    const result = [[rStart, cStart]];
    const directions = [[0, 1], [1, 0], [0, -1], [-1, 0]]; // right, down, left, up
    let steps = 1;
    let dir = 0;
    let r = rStart, c = cStart;
    
    while (result.length < rows * cols) {
        // Move in current direction for 'steps' times
        for (let i = 0; i < 2; i++) {
            const [dr, dc] = directions[dir];
            
            for (let j = 0; j < steps; j++) {
                r += dr;
                c += dc;
                
                // Add if within bounds
                if (r >= 0 && r < rows && c >= 0 && c < cols) {
                    result.push([r, c]);
                }
            }
            
            dir = (dir + 1) % 4;  // Change direction
        }
        
        steps++;  // Increase steps after every 2 directions
    }
    
    return result;
};
```
