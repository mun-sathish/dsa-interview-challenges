### [Valid Sudoku](https://leetcode.com/problems/valid-sudoku/)

- Check each row, column, and 3Ã—3 sub-box for duplicates
- Use hash sets to track seen numbers
- Box index formula: (row / 3) * 3 + (col / 3)
- Only validate filled cells (ignore '.')

#### Visual Explanation

**Validation Areas:**

- Col: 7
- Box: (4/3)*3 + (7/3) = 1*3 + 2 = 5

```js copy
/**
 * @param {character[][]} board
 * @return {boolean}
 */
var isValidSudoku = function(board) {
    const rows = Array.from({ length: 9 }, () => new Set());
    const cols = Array.from({ length: 9 }, () => new Set());
    const boxes = Array.from({ length: 9 }, () => new Set());
    
    for (let i = 0; i < 9; i++) {
        for (let j = 0; j < 9; j++) {
            const num = board[i][j];
            
            if (num === '.') continue;
            
            // Calculate box index
            const boxIndex = Math.floor(i / 3) * 3 + Math.floor(j / 3);
            
            // Check if number already exists
            if (rows[i].has(num) || cols[j].has(num) || boxes[boxIndex].has(num)) {
                return false;
            }
            
            // Add number to sets
            rows[i].add(num);
            cols[j].add(num);
            boxes[boxIndex].add(num);
        }
    }
    
    return true;
};
```

##### Complexity

- Time: O(1) since board is always 9Ã—9
- Space: O(1) constant space for sets
