### [Lucky Numbers in a Matrix](https://leetcode.com/problems/lucky-numbers-in-a-matrix/)

- Lucky number = minimum in its row AND maximum in its column
- Find min in each row, check if it's max in its column
- Can have at most one lucky number
- Two-pass solution or optimized single pass

```js copy
/**
 * @param {number[][]} matrix
 * @return {number[]}
 */
var luckyNumbers = function(matrix) {
    const m = matrix.length;
    const n = matrix[0].length;
    const result = [];
    
    // Find minimum in each row
    for (let i = 0; i < m; i++) {
        let minVal = matrix[i][0];
        let minCol = 0;
        
        for (let j = 1; j < n; j++) {
            if (matrix[i][j] < minVal) {
                minVal = matrix[i][j];
                minCol = j;
            }
        }
        
        // Check if this min is also max in its column
        let isMaxInCol = true;
        for (let k = 0; k < m; k++) {
            if (matrix[k][minCol] > minVal) {
                isMaxInCol = false;
                break;
            }
        }
        
        if (isMaxInCol) {
            result.push(minVal);
        }
    }
    
    return result;
};
```
