### [Permutations](https://leetcode.com/problems/permutations/)

- Use backtracking to generate all permutations
- Track used elements to avoid duplicates
- Add element, recurse, then backtrack
- Add to result when permutation is complete

```js copy
/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var permute = function(nums) {
    const result = [];
    const current = [];
    const used = new Set();
    
    function backtrack() {
        if (current.length === nums.length) {
            result.push([...current]);
            return;
        }
        
        for (let i = 0; i < nums.length; i++) {
            if (used.has(i)) continue;
            
            current.push(nums[i]);
            used.add(i);
            
            backtrack();
            
            current.pop();
            used.delete(i);
        }
    }
    
    backtrack();
    return result;
};
```

##### Complexity

- Time: O(n! * n) where n is length of array
- Space: O(n) for recursion stack
