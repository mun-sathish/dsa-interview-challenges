### [Surrounded Regions](https://leetcode.com/problems/surrounded-regions/)

- **Matrix DFS/BFS Pattern**: Capture surrounded regions
- Start DFS/BFS from border 'O's (they can't be captured)
- Mark border-connected 'O's as safe
- Flip remaining 'O's to 'X' (they're surrounded)

```js copy
/**
 * @param {character[][]} board
 * @return {void} Do not return anything, modify board in-place instead.
 */
var solve = function(board) {
    if (!board || board.length === 0) return;
    
    const rows = board.length;
    const cols = board[0].length;
    
    // DFS to mark border-connected 'O's as safe
    function dfs(i, j) {
        if (i < 0 || i >= rows || j < 0 || j >= cols || board[i][j] !== 'O') {
            return;
        }
        
        board[i][j] = 'S'; // Mark as safe
        
        dfs(i + 1, j);
        dfs(i - 1, j);
        dfs(i, j + 1);
        dfs(i, j - 1);
    }
    
    // Step 1: Mark all border-connected 'O's as safe
    // Check first and last rows
    for (let j = 0; j < cols; j++) {
        if (board[0][j] === 'O') dfs(0, j);
        if (board[rows - 1][j] === 'O') dfs(rows - 1, j);
    }
    
    // Check first and last columns
    for (let i = 0; i < rows; i++) {
        if (board[i][0] === 'O') dfs(i, 0);
        if (board[i][cols - 1] === 'O') dfs(i, cols - 1);
    }
    
    // Step 2: Flip surrounded 'O's to 'X' and restore safe 'S' to 'O'
    for (let i = 0; i < rows; i++) {
        for (let j = 0; j < cols; j++) {
            if (board[i][j] === 'O') {
                board[i][j] = 'X'; // Surrounded, flip it
            } else if (board[i][j] === 'S') {
                board[i][j] = 'O'; // Safe, restore it
            }
        }
    }
};
```

##### Complexity

- Time: O(m * n) where m is rows and n is columns
- Space: O(m * n) for recursion stack in worst case
